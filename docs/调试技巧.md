# 服务维护

临时关闭服务
sudo systemctl stop doly-daemon.service
永久关闭，禁用服务(关闭自动启动)
sudo systemctl disable doly-daemon.service
恢复开机自动启动
sudo systemctl enable doly-daemon.service

禁用服务
sudo systemctl disable doly-daemon.service
sudo systemctl stop doly-daemon.service
sudo systemctl stop eyeengine-service.service

sudo systemctl status doly-audio-player.service

sudo systemctl restart eyeengine-service.service

sudo rm /etc/systemd/system/drive-service.service
sudo systemctl mask drive-service.service
sudo systemctl stop drive-service.service
sudo systemctl daemon-reload

## 恢复：
sudo systemctl unmask drive-service.service
cd /home/pi/dolydev/libs/drive/scripts
sudo cp /home/pi/dolydev/libs/drive/scripts/drive-service.service /etc/systemd/system/

sudo systemctl daemon-reload
sudo systemctl enable drive-service.service
sudo systemctl start drive-service.service

## 禁用探索和悬崖传感器
在config/exploration.yaml 中把cliff禁用然后重启doly-daemon.service
sudo systemctl restart doly-daemon.service
